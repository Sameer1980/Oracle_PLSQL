SELECT
LAST_NAME,
TO_CHAR(HIRE_DATE,'DD-MON-YY')"HIRE_DATE",
TO_CHAR(SYSDATE,'DD-MON-YY') "SYSDATE",
(
CASE 
WHEN SYSDATE-TO_YMINTERVAL('15-0')>=HIRE_DATE THEN '15 YEARS OF SERVICE' 
WHEN SYSDATE-TO_YMINTERVAL('10-0')>=HIRE_DATE THEN '10 YEARS OF SERVICE' 
WHEN SYSDATE-TO_YMINTERVAL('05-0')>=HIRE_DATE THEN '5 YEARS OF SERVICE' ELSE 'MAYBE NEXT YEAR!'
END
)
FROM EMPLOYEES
ORDER BY 4



SELECT
LAST_NAME,
TO_CHAR(HIRE_DATE,'DD-MON-YY')"HIRE_DATE",
TO_CHAR(SYSDATE,'DD-MON-YY') "SYSDATE",
(
CASE 
WHEN TRUNC(MONTHS_BETWEEN(SYSDATE,HIRE_DATE)/12)>=15 THEN '15 YEARS OF SERVICE' 
WHEN TRUNC(MONTHS_BETWEEN(SYSDATE,HIRE_DATE)/12)>=10 THEN '10 YEARS OF SERVICE'  
WHEN TRUNC(MONTHS_BETWEEN(SYSDATE,HIRE_DATE)/12)>=05 THEN '05 YEARS OF SERVICE' 
ELSE
'MAYBE NEXT YEAR!'
END
)
FROM EMPLOYEES
ORDER BY "HIRE_DATE"
