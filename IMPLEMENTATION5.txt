CREATE OR REPLACE TRIGGER AUDIT_EMPVALUES 
AFTER INSERT OR UPDATE OR DELETE ON DEMO_EMP REFERENCING OLD AS OLD NEW AS NEW
FOR EACH ROW
BEGIN
INSERT INTO AUDIT_EMP(USER_NAME,TIME_STAMP,EMPID,OLD_LAST_NAME,NEW_LAST_NAME,OLD_TITLE,NEW_TITLE,OLD_SALARY,NEW_SALARY)
VALUES(USER,SYSDATE,:OLD.EMPLOYEE_ID,:OLD.LAST_NAME,:NEW.LAST_NAME,:OLD.JOB_ID,:NEW.JOB_ID,:OLD.SALARY,:NEW.SALARY); 
END;
