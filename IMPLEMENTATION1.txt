
|----------- PROCEDURE CALLED FROM TRIGGER---------------|



I) FIRST 'GRANT ADMINISTER DATABASE TRIGGER TO PIS' FROM ADM.

II)THEN CREATE A TABLE 'AUDIT_TABLE' :-

USER_NAME VARCHAR2(32),LOG_DATE DATE,ACTION VARCHAR2(35); 

III)THEN CREATE A PROCEDURE:-

CREATE OR REPLACE PROCEDURE AUDIT_PROC(USER_ID VARCHAR2)
AS
BEGIN
INSERT INTO AUDIT_TABLE (USER_NAME,LOG_DATE,ACTION) VALUES(USER,SYSDATE,'Logging On');
END;




IV)THEN CALL THE PROCEDURE FROM THE TRIGGER:-

CREATE OR REPLACE TRIGGER DATABASE_TRIG
AFTER LOGON ON DATABASE
begin
AUDIT_PROC(ORA_LOGIN_USER);
end;
