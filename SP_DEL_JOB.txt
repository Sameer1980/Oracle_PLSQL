CREATE OR REPLACE PROCEDURE SP_DEL_JOB (P_JOBID VARCHAR2) AS

V_COUNT NUMBER(4);
NO_DATA_FOUND EXCEPTION;
BEGIN

SELECT COUNT(*) INTO V_COUNT FROM JOBS WHERE JOB_ID=P_JOBID;

IF V_COUNT=0 THEN 
RAISE NO_DATA_FOUND;
END IF;


DELETE FROM JOBS WHERE JOB_ID=P_JOBID;
COMMIT;
EXCEPTION
WHEN NO_DATA_FOUND THEN
DBMS_OUTPUT.PUT_LINE('ERR-0102.... JOB NOT FOUND');

END;
