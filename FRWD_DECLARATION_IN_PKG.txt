CREATE OR REPLACE PACKAGE FORWARD_PKG AS
PROCEDURE ANNUAL_SAL(P_EMPID VARCHAR2);
END;


CREATE OR REPLACE PACKAGE BODY FORWARD_PKG AS
PROCEDURE TAX_CALC(P_EMPID IN VARCHAR2,REC_AMT OUT  NOCOPY NUMBER);

PROCEDURE ANNUAL_SAL(P_EMPID VARCHAR2) AS
ANN_SAL NUMBER;
BEGIN
TAX_CALC(P_EMPID,ANN_SAL);
DBMS_OUTPUT.PUT_LINE('Annual Salary: '||ANN_SAL);
END;

PROCEDURE TAX_CALC(P_EMPID IN VARCHAR2,REC_AMT OUT NOCOPY NUMBER) AS
BEGIN
SELECT (SALARY * 12) - (SALARY * 12 * 0.25) INTO REC_AMT FROM EMPLOYEES 
WHERE EMPLOYEE_ID=P_EMPID;
END;

END;
