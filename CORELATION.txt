SELECT A.EMPLOYEE_ID,
INITCAP(FIRST_NAME||' '||LAST_NAME) "NAME",
JOB_ID
FROM EMPLOYEES A
WHERE 
2<=(SELECT COUNT(*) FROM JOB_HISTORY WHERE EMPLOYEE_ID=A.EMPLOYEE_ID) 


SELECT
 A.EMPLOYEE_ID,
 INITCAP(A.FIRST_NAME||' '||A.LAST_NAME) "NAME",
 A.JOB_ID,
 B.DEPARTMENT_NAME "DEPARTMENT"
 FROM EMPLOYEES A,
 DEPARTMENTS B
 WHERE 
 A.DEPARTMENT_ID=B.DEPARTMENT_ID(+)
 AND
 EXISTS(SELECT 'X' FROM EMPLOYEES WHERE MANAGER_ID=A.EMPLOYEE_ID)
