CREATE 
OR 
REPLACE 
FUNCTION
GET_ANNUAL_COMP 
(SAL NUMBER,COMM NUMBER) 
RETURN NUMBER
IS

ANNUAL_COMP NUMBER(7);

BEGIN
ANNUAL_COMP:=(SAL*12)+(NVL(COMM,0)*SAL*12);
RETURN(ANNUAL_COMP);
END;


SELECT EMPLOYEE_ID,LAST_NAME,GET_ANNUAL_COMP(SALARY,COMMISSION_PCT)"Annual Compensation"  FROM EMPLOYEES WHERE DEPARTMENT_ID=30;
