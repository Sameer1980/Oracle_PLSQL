SELECT
INITCAP(FIRST_NAME||' '||LAST_NAME) "NAME",
TO_CHAR(TO_DATE(HIRE_DATE,'DD-MM-RR'),'DD-MON-YYYY') "HIRE DATE",
(CASE EXTRACT(YEAR FROM HIRE_DATE) WHEN TO_NUMBER('1998') THEN 'Needs Review'
ELSE 
'not this year!!' END) "REVIEW"
FROM 
EMPLOYEES
ORDER BY TO_DATE(HIRE_DATE,'DD-MM-RR');


SELECT EXTRACT(YEAR FROM SYSDATE) ,
EXTRACT (MONTH FROM SYSDATE),
EXTRACT(DAY FROM SYSDATE),
EXTRACT(HOUR FROM LOCALTIMESTAMP),
EXTRACT(MINUTE FROM LOCALTIMESTAMP),
EXTRACT(SECOND FROM LOCALTIMESTAMP)
FROM DUAL;
