DECLARE
CURSOR DATE_CUR(MY_DATE DATE) IS
SELECT * FROM EMPLOYEES WHERE TO_DATE(HIRE_DATE,'DD-MM-RR')>TO_DATE(MY_DATE,'DD-MM-RR');
THIS_DATE DATE:='&TH_DT';

BEGIN
FOR REC IN DATE_CUR(THIS_DATE) LOOP
DBMS_OUTPUT.PUT_LINE('EMPLOYEE ID: '||REC.EMPLOYEE_ID||'| '||'FIRST NAME: '||REC.FIRST_NAME||' |'||'LAST NAME: '||REC.LAST_NAME||'| '||'SALARY: '||
REC.SALARY||'| '||'HIRE DATE: '||REC.HIRE_DATE||'| '||'COMMISSION PCT: '||REC.COMMISSION_PCT||' '||' |'||'DEPARTMENT ID: '||REC.DEPARTMENT_ID);
END LOOP;
END;
